function shuffleArray (arr) {
  var shuffled = [], indexes = [], item, num;

  while (indexes.length < arr.length) {
    num = Math.floor(Math.random() * 59);
    if (indexes.indexOf(num) === -1) {
      indexes.push(num);
    }
  }

  for (num = 0;num < indexes.length;num++) {
    shuffled[indexes[num]] = arr[num];
  }

  return shuffled;
}
