function insertionSort (list) {
  var outerIdx, length = list.length, innerIdx, currItem;
  
  for (outerIdx = 1;outerIdx < length;outerIdx++) {
    currItem = list[outerIdx];
    for (innerIdx = outerIdx - 1;innerIdx >= 0 && list[innerIdx] > currItem;innerIdx--) {
      list[innerIdx + 1] = list[innerIdx];
    }
    list[innerIdx + 1] = currItem;
  }
}
